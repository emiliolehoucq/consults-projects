Create variables for maps

Define variables

data <- data %>%
  mutate(
    avg_diff_stock = if_else(
      if_any(c(diff_stock_2010, diff_stock_2015, diff_stock_2020), is.na),
      NA,
      rowMeans(select(., diff_stock_2010, diff_stock_2015, diff_stock_2020))
    ),
    avg_ratio_stock = if_else(
      if_any(c(ratio_stock_2010, ratio_stock_2015, ratio_stock_2020), is.na),
      NA,
      rowMeans(select(., ratio_stock_2010, ratio_stock_2015, ratio_stock_2020))
    )
  )

p5 <- quantile(data$avg_ratio_stock, probs = 0.05, na.rm = TRUE)
p50 <- median(data$avg_ratio_stock, na.rm = TRUE)
p90 <- quantile(data$avg_ratio_stock, probs = 0.9, na.rm = TRUE)

data <- data %>%
  mutate(
    avg_net_pos = if_else(avg_diff_stock > 0, 1, 0),
    stock_2020_net_pos = ifelse(diff_stock_2020 > 0, 1, 0),
    diff_balance = case_when(
      is.na(diff_stock_2010) | is.na(diff_stock_2015) | is.na(diff_stock_2020) ~ NA,
      diff_stock_2010 <= 0 & diff_stock_2015 <= 0 & diff_stock_2020 <= 0 ~ "Always more emigrants from the US",
      diff_stock_2010 > 0 & diff_stock_2015 > 0 & diff_stock_2020 > 0 ~ "Always more immigrants to the US",
      TRUE ~ "Balance varies by year"
    ),
    stock_2020_category_three = case_when(
      stock_2020_destination_to_origin > stock_2020_origin_to_destination &
        stock_2020_origin_to_destination < stock_2020_destination_to_origin * 0.9 ~ "More emigrants from the US",
      pmin(stock_2020_origin_to_destination, stock_2020_destination_to_origin) /
        pmax(stock_2020_origin_to_destination, stock_2020_destination_to_origin) >= 0.90 ~ "About the same",
      stock_2020_origin_to_destination > stock_2020_destination_to_origin &
        stock_2020_destination_to_origin < stock_2020_origin_to_destination * 0.9 ~ "More immigrants from the US"
    ),
    stock_2020_category_four = case_when(
      stock_2020_origin_to_destination > stock_2020_destination_to_origin &
        stock_2020_destination_to_origin < stock_2020_origin_to_destination * 0.9 &
        stock_2020_destination_to_origin >= stock_2020_origin_to_destination * 0.1 ~ "More immigrants to the US",
      stock_2020_origin_to_destination > stock_2020_destination_to_origin &
        stock_2020_destination_to_origin < stock_2020_origin_to_destination * 0.1 ~ "Many more immigrants to the US",
      pmin(stock_2020_origin_to_destination, stock_2020_destination_to_origin) /
        pmax(stock_2020_origin_to_destination, stock_2020_destination_to_origin) >= 0.90 ~ "About the same",
      stock_2020_destination_to_origin > stock_2020_origin_to_destination &
        stock_2020_origin_to_destination < stock_2020_destination_to_origin * 0.9 ~ "More emigrants from the US"
    ),
  avg_ratio_stock_category = case_when(
    is.na(avg_ratio_stock) ~ NA,
    avg_ratio_stock < p5 ~ "Below 5th percentile",
    avg_ratio_stock >= p5 & avg_ratio_stock < p50 ~ "Between 5th percentile and median",
    avg_ratio_stock >= p50 & avg_ratio_stock < p90 ~ "Between median and 90th percentile",
    TRUE ~ "Above 90th percentile"
  )
  )
