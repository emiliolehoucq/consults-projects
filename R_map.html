Map of avg_net_pos

Convert avg_net_pos to factor

net_positive_label <- "More immigrants to the US"
net_negative_label <- "More emigrants from the US"
na_label <- "Data not available"

data$avg_net_pos <- factor(
  data$avg_net_pos,
  levels = c(0, 1),
  labels = c(net_negative_label, net_positive_label)
  )

data$avg_net_pos <- addNA(data$avg_net_pos)
levels(data$avg_net_pos)[is.na(levels(data$avg_net_pos))] <- na_label

data$avg_net_pos <- factor(
  data$avg_net_pos,
  levels = c(net_positive_label, net_negative_label, na_label)
  )

Create map

plot_avg_net_pos <- ggplot(data) +
  aes(x = long, y = lat, group = group) +
  geom_polygon(aes(fill = avg_net_pos), color = "gray60", lwd=.15) +
  scale_fill_manual(
      name = NULL,
      values = setNames(
        c("gray30", "gray80", "white"),
        c(net_negative_label, net_positive_label, na_label)
      )
  ) +
  scale_y_continuous(limits=c(-65,NA)) +
  scale_x_continuous(limits=c(NA, 180), expand=c(0,0)) + 
  coord_map("mollweide") + 
  map_theme

Save

ggsave("../visualizations/avg_net_pos.pdf", plot = plot_avg_net_pos, width = 11, height = 8.5)
